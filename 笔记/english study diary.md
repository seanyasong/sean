        ad-：a-,ac-,ad-,af-,ag-,al-,an-,ap-,ar-,as-,at-等。
        ab-：abs,aps,as,au,a
        con-:cog,col,com,cor,co
        in-:im,il,ir,ig
        en-:em
        syn-:sym,syl,sy
        ob-:oc,of,op,o
        sub-:suc,suf,sup,sug,sur,sum,sus,su
        trans-:tran,tra
        dis-:dif,dir,di
        ex-:ef,e,ec,es

        adhere  坚持&依附
